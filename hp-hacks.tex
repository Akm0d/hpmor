\def\sbreak{\mbox{}\nobreak\hfill\mbox{}\allowbreak\rule{.60\textwidth}{.0pt}\par%
  \vskip 0pt plus 2\baselineskip\noindent{%
    \parbox[c][0pt][c]{\textwidth}{%
      \hfil \hbox{\lower14pt\hbox{\normalsize\Stars}}%
    }%
  }}
